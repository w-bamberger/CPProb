00140       {
00141          // The output value can be negative, because of rounding errors in
00142          // the difference. To avoid (impossible) negative evidence,
00143          // I check for that throughout this method.
00144 
00145          float sumEvidencePlusC = N + size_;

00147 
00148          const float* in = probabilities;
00149          float* out = evidence;
00150 
00151          // All but the last element in evidence
00152          for ( ; out != evidence + size_ - 1; ++in, ++out )
00153          {
00154             *out = sumEvidencePlusC * *in - 1; // -1, because a = 1/size_ here
00155             *out = *out < 0 ? 0.0f : *out;

00562       Opinion<size_>
00563       beliefDiscountingBy( float discountingFactor ) const
00564       {
00565          /*
00566           * = Multinomial discounting in belief representation (LaTeX code) =
00567           *
00568           *   b^{A:B}(x_i) = b^A_B * b^B(x_i)
00569           *        u^{A:B} = 1 - b^A_B * (1 - u^B_x)
00570           *        a^{A:B} = a^B
00571           *
00572           * This is how Jøsang describes it. Note that only one factor of the
00573           * trust opinion is necessary. I call it the discounting factor.
00574           *
00575           * This implementation of an opinion works in evidence representation.
00576           * So an algorithm is necessary that does the above computations with
00577           * the evidence vectors. This is described next and then implemented
00578           * below.
00579           *
00580           * = Multinomial discounting in evidence representation (LaTeX code) =
00581           *
00582           * From the uncertainty of the discounted opinion, the sum S^{A:B} of
00583           * the discounted evidence can be derived:
00584           *
00585           *   S^{A:B} + C = \frac{C}{S^B * (1 - b^A_B) + C} * (S^B + C) .
00586           *
00587           * This gives us the relation
00588           *
00589           *   R = \frac{S^{A:B} + C}{S^B + C}
00590           *
00591           * between input evidence and the discounted evidence.
00592           *
00593           * From the first belief equation, the discounting of the evidence
00594           * values can be derived now:
00595           *
00596           *    \frac{r^{A:B}(x_i)}{S^{A:B} + C} =
00597           *        b^A_B * \frac{r^B(x_i)}{S^B + C} .
00598           *
00599           * When resolving for the discounted evidence value, the computation
00600           * of the belief discounting in evidence representation results in
00601           *
00602           *    r^{A:B}(x_i) = b^A_B * r^B(x_i) * R .
00603           *
00604           * This is implemented below.
00605           */
00606 
00607          float sumOfEvidence = std::accumulate(evidence_, evidence_ + size_, 0);
00608          float evidenceRelation = sizef_ / (sumOfEvidence * (1.0f
00609                   - discountingFactor) + sizef_);
00610 
00611          float scalingFactor = discountingFactor * evidenceRelation;
00612          float outEvidence[size_];
00613          std::transform(evidence_, evidence_ + size_, outEvidence,
00614                   std::bind2nd(std::multiplies<float>(), scalingFactor));
00615 
00616          return Opinion<size_>::fromEvidence(outEvidence, baseRate_);
00617       }
